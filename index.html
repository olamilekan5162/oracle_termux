Step 0: Install Termux

Download and install Termux from F-Droid (recommended) or from the official GitHub releases.

Step 1: Update and Upgrade Packages

pkg update && pkg upgrade

Step 2: Install Neovim

pkg install neovim

Step 3: Install Vim Plug

curl -fLo ~/.local/share/nvim/site/autoload/plug.vim --create-dirs \
https://raw.githubusercontent.com/junegunn/vim-plug/master/plug.vim

Step 4: Setup Config Directory

mkdir -p ~/.config/nvim
touch ~/.config/nvim/init.vim

Step 5: Open Config File

nano ~/.config/nvim/init.vim

Step 6: Add Neovim Config

call plug#begin('~/.vim/plugged')
" File Explorer
Plug 'preservim/nerdtree'
" Syntax Highlighting
Plug 'nvim-treesitter/nvim-treesitter', {'do': ':TSUpdate'}
" Autocompletion
Plug 'neoclide/coc.nvim', {'branch': 'release'}
" Status Line
Plug 'vim-airline/vim-airline'
Plug 'vim-airline/vim-airline-themes'
Plug 'honza/vim-snippets'
Plug 'SirVer/ultisnips'
" Theme
Plug 'morhetz/gruvbox'
" Git Integration
Plug 'tpope/vim-fugitive'
call plug#end()

" Automatically open NERDTree
autocmd StdinReadPre * let s:std_in=1
autocmd VimEnter * if argc() == 1 && isdirectory(argv()[0]) | execute 'NERDTree' argv()[0] | wincmd p | endif

syntax enable
colorscheme gruvbox
set background=dark

inoremap   pumvisible() ? coc#pum#confirm() : "\u\"
inoremap   coc#pum#visible() ? coc#pum#confirm() : "\u\"

let g:airline#extensions#tabline#enabled = 1
let g:airline_theme='gruvbox'

nnoremap  :NERDTreeToggle

set number
set relativenumber
set tabstop=4
set shiftwidth=4
set expandtab
set mouse=a
set clipboard=unnamedplus

lua << EOF
require'nvim-treesitter.configs'.setup {
    highlight = {
        enable = true,
        additional_vim_regex_highlighting = false,
    },
}
EOF

Step 7: Save and Exit

Press CTRL + O, Enter, then CTRL + X

Step 8: Install Plugins

nvim
:PlugInstall

Step 9: Setup coc-settings.json

mkdir -p ~/.config/nvim/coc-settings.json
nano ~/.config/nvim/coc-settings.json

Step 10: Add coc-settings.json Config

{
  "prettier.enable": true,
  "prettier.singleQuote": true,
  "prettier.trailingComma": "all",
  "prettier.semi": true,
  "prettier.useTabs": false,
  "coc.preferences.formatOnSave": true,
  "prettier.arrowParens": "always",
  "coc.preferences.formatOnSaveFiletypes": ["javascript", "typescript", "json", "css", "html"],
  "suggest.acceptSuggestionOnEnter": "always",
  "snippets.ultisnips.pythonPrompt": false,
  "snippets.userSnippetsDirectory": "~/.config/coc/ultisnips",
  "suggest.noselect": false,
  "coc.preferences.snippetIndicator": "‚óè",
  "snippets.enable": true,
  "suggest.enablePreselect": true
}

Step 11: Install coc Extensions

:CocInstall coc-tsserver coc-html coc-css coc-json
:CocInstall coc-prettier
:CocInstall coc-eslint
:CocInstall coc-snippets
:CocCommand snippets.openSnippetFiles
